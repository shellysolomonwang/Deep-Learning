Decentralized Optimization¶

The i-th node repeats:

  1. Compute gradient $g_i$ using its local data and current parameters $w_i$
  2. Pull the parameters from its neighbors, denote${w_k}$
  3. $w_i =$ weighted average of $w_i$ and ${w_k}$
  4. $w_i = w_i - \alpha * g_i$

class Worker:
    def __init__ (self, x, y):
        self.x = x # sbyd local feature matrix 
        self.y = y # sby1 local label matrix
        self.s = x.shape[0] # number of local samples
        self.d = x.shape[1] # number of features 
        self.w = numpy.zeros((self.d, 1)) # dby1model parameter vector
        self.neighbors = [] # list of neighbors
        
    # Set the model parameters to the latest
    def set_param(self , w): 
        self.w = w
        
    # Set neighbors
    def set_neighbors(self, neighbors):
        self.neighbors.extend(neighbors)
        
#     # Get neighbors
#     def get_neighbors(self):
#         return self.neighbors
    
#     def get_w(self):
#         return self.w
    
    # Compute the local loss
    def loss(self):
        yx = numpy.multiply(self.y, self.x) # sbyd matrix 
        yxw = numpy.dot(yx, self.w) # sby1 matrix
        vec1 = numpy.exp(-yxw) # sby1 matrix
        vec2 = numpy.log(1 + vec1) # sby1 matrix
        return numpy.sum(vec2) # loss function

    # Compute the local gradient
    def gradient(self):
        yx = numpy.multiply(self.y, self.x) # sbyd matrix
        yxw = numpy.dot(yx, self.w) # sby1 matrix
        vec1 = numpy.exp(yxw) # sby1 matrix
        vec2 = numpy.divide(yx, 1+vec1) # sbyd matrix
        g = -numpy.sum(vec2, axis=0).reshape(self.d, 1) # d-by-1 matrix
        return g

import math

# Create m worker nodes
def create_workers(m, x, y):
    n, d = x.shape
    s = math.floor(n/m)
    workers = []
    
    
    # assign s training samples to each worker
    for i in range(m):
        indices = list(range(i*s, (i+1)*s))
        worker = Worker(x[ indices , :] , y[ indices , :]) # class Worker
        workers.append(worker)
    
    # set neighbors
    workers[0].set_neighbors([1,2])
    workers[1].set_neighbors([0,2,3])
    workers[2].set_neighbors([0,1,4])
    workers[3].set_neighbors([1,4])
    workers[4].set_neighbors([2,3,5,6])
    workers[5].set_neighbors([4,6])
    workers[6].set_neighbors([4,5])
    return workers, s, d

def fit_model(weight_scheme):
    m = 7 # number of worker nodes
    workers, s, d = create_workers(m, x_train, y_train)
    
    lam = 1E-3 # regularization parameter 
#     alpha = 1 # learning rate / stepsize for agd
#     beta = 0.9 # momentum parameter
    max_epoch = 50 # number of epochs
    worker_alpha = 1E-2
    obj_values = []
    
        
    for t in range(max_epoch):
        
        w_avg = []
        loss = []
        
        for i in range(m):
            
            # step 1: Compute gradient    using its local data and current parameters  
            g_i = workers[i].gradient()
            l_i = workers[i].loss()
            w_i = workers[i].w
            
            # step 2: pull the parameters from its neighbors denot w_k
            w_k = []
            for neighbor in workers[i].neighbors:
                w_k.append(workers[neighbor].w)
            
            # step 3: compute weighted average of w_i & w_k
            if weight_scheme == 'AVG':
                w_k.append(w_i)
                w_i = numpy.mean(w_k)
            else:
                wei = [1 for k in w_k]
                w_k.append(w_i)
                wei.append(2)
                #print(len(wei), len(w_k))
                w_i = numpy.average(w_k, axis = 0, weights = wei)
            
            # step 4: update w_i with current w_i and g_i
            w_i -= worker_alpha * g_i
            workers[i].w = w_i
            
            w_avg.append(w_i)
            loss.append(l_i)
        
        w_avg = numpy.sum(w_avg) / m
        reg = lam/2 * numpy.sum(w_avg*w_avg) # scalar
        loss = numpy.sum(loss) / (s*m)
        obj = loss + reg
        obj_values.append(obj)
        print('Objective function value at epoch '+str(t)+' = ' + str(obj))
    
    return obj_values
